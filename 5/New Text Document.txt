#include <iostream>
#include<conio.h>
#include<cstdlib>
using namespace std;

int diagonal(int* (ptr), int col)
{
	int x = 0;
	for (int i = 0; i <= col; i++)
	{
		x = *(ptr + i * 3 + i);
	}
	return x;
}

int main()
{
	srand(time_t(NULL));
	int arr[3][3];
	int *ptr = &arr[0][0];
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			*(ptr+i*3+j) = rand() % 16;
		}
	}
	cout << "The is the 2D 3x3 Array generated by the Computer" << endl;
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			cout << *(ptr + i * 3 + j) << "\t";
		}
		cout << endl;
	}
	cout << "Now we will pass the Array to the function 'diagonal' and seperate diagonal values from the function.\n";
	for (int i = 0; i < 3; i++)
	{
		cout << diagonal(ptr, i) << endl;
	}
	system("Pause");
	return 0;
}
